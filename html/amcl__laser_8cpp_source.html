143  {
144  obs_range = data->ranges[i][0];
145  obs_bearing = data->ranges[i][1];
146 
147  // Compute the range according to the map
148  map_range = map_calc_range(self->map, pose.v[0], pose.v[1],
149  pose.v[2] + obs_bearing, data->range_max);
150  pz = 0.0;
151 
152  // Part 1: good, but noisy, hit
153  z = obs_range - map_range;
154  pz += self->z_hit * exp(-(z * z) / (2 * self->sigma_hit * self->sigma_hit));
155 
156  // Part 2: short reading from unexpected obstacle (e.g., a person)
157  if(z < 0)
158  pz += self->z_short * self->lambda_short * exp(-self->lambda_short*obs_range);
159 
160  // Part 3: Failure to detect obstacle, reported as max-range
161  if(obs_range == data->range_max)
162  pz += self->z_max * 1.0;
163 
164  // Part 4: Random measurements
165  if(obs_range < data->range_max)
166  pz += self->z_rand * 1.0/data->range_max;
167 
168  // TODO: outlier rejection for short readings
169 
170  assert(pz <= 1.0);
171  assert(pz >= 0.0);
172  // p *= pz;
173  // here we have an ad-hoc weighting scheme for combining beam probs
174  // works well, though...
175  p += pz*pz*pz;
176  }
177 
178  sample->weight *= p;

180  }
