74  void MoveSlowAndClear::runBehavior()
75  {
76  if(!initialized_)
77  {
78  ROS_ERROR("This recovery behavior has not been initialized, doing nothing.");
79  return;
80  }
81  ROS_WARN("Move slow and clear recovery behavior started.");
82  tf::Stamped<tf::Pose> global_pose, local_pose;
83  global_costmap_->getRobotPose(global_pose);
84  local_costmap_->getRobotPose(local_pose);
85 
86  std::vector<geometry_msgs::Point> global_poly, local_poly;
87  geometry_msgs::Point pt;
88 
89  for(int i = -1; i <= 1; i+=2)
90  {
91  pt.x = global_pose.getOrigin().x() + i * clearing_distance_;
92  pt.y = global_pose.getOrigin().y() + i * clearing_distance_;
93  global_poly.push_back(pt);
94 
95  pt.x = global_pose.getOrigin().x() + i * clearing_distance_;
96  pt.y = global_pose.getOrigin().y() + -1.0 * i * clearing_distance_;
97  global_poly.push_back(pt);
98 
99  pt.x = local_pose.getOrigin().x() + i * clearing_distance_;
100  pt.y = local_pose.getOrigin().y() + i * clearing_distance_;
101  local_poly.push_back(pt);
102 
103  pt.x = local_pose.getOrigin().x() + i * clearing_distance_;
104  pt.y = local_pose.getOrigin().y() + -1.0 * i * clearing_distance_;
105  local_poly.push_back(pt);
106  }
107 
108  //clear the desired space in the costmaps
109  std::vector<boost::shared_ptr<costmap_2d::Layer> >* plugins = global_costmap_->getLayeredCostmap()->getPlugins();
110  for (std::vector<boost::shared_ptr<costmap_2d::Layer> >::iterator pluginp = plugins->begin(); pluginp != plugins->end(); ++pluginp) {
111  boost::shared_ptr<costmap_2d::Layer> plugin = *pluginp;
112  if(plugin->getName().find("obstacles")!=std::string::npos){
113  boost::shared_ptr<costmap_2d::ObstacleLayer> costmap;
114  costmap = boost::static_pointer_cast<costmap_2d::ObstacleLayer>(plugin);
115  costmap->setConvexPolygonCost(global_poly, costmap_2d::FREE_SPACE);
116  }
117  }
118 
119  plugins = local_costmap_->getLayeredCostmap()->getPlugins();
120  for (std::vector<boost::shared_ptr<costmap_2d::Layer> >::iterator pluginp = plugins->begin(); pluginp != plugins->end(); ++pluginp) {
121  boost::shared_ptr<costmap_2d::Layer> plugin = *pluginp;
122  if(plugin->getName().find("obstacles")!=std::string::npos){
123  boost::shared_ptr<costmap_2d::ObstacleLayer> costmap;
124  costmap = boost::static_pointer_cast<costmap_2d::ObstacleLayer>(plugin);
125  costmap->setConvexPolygonCost(local_poly, costmap_2d::FREE_SPACE);
126  }
127  }
128 
129  //lock... just in case we're already speed limited
130  boost::mutex::scoped_lock l(mutex_);
131 
132  //get the old maximum speed for the robot... we'll want to set it back
133  if(!limit_set_)
134  {
135  if(!planner_nh_.getParam("max_trans_vel", old_trans_speed_))
136  {
137  ROS_ERROR("The planner %s, does not have the parameter max_trans_vel", planner_nh_.getNamespace().c_str());
138  }
139 
140  if(!planner_nh_.getParam("max_rot_vel", old_rot_speed_))
141  {
142  ROS_ERROR("The planner %s, does not have the parameter max_rot_vel", planner_nh_.getNamespace().c_str());
143  }
144  }
145 
146  //we also want to save our current position so that we can remove the speed limit we impose later on
147  speed_limit_pose_ = global_pose;
148 
149  //limit the speed of the robot until it moves a certain distance
150  setRobotSpeed(limited_trans_speed_, limited_rot_speed_);
151  limit_set_ = true;
152  distance_check_timer_ = private_nh_.createTimer(ros::Duration(0.1), &MoveSlowAndClear::distanceCheck, this);
153  }

155  double MoveSlowAndClear::getSqDistance()
156  {
157  tf::Stamped<tf::Pose> global_pose;
158  global_costmap_->getRobotPose(global_pose);
159  double x1 = global_pose.getOrigin().x();
160  double y1 = global_pose.getOrigin().y();
161 
162  double x2 = speed_limit_pose_.getOrigin().x();
163  double y2 = speed_limit_pose_.getOrigin().y();
164 
165  return (x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1);
166  }

168  void MoveSlowAndClear::distanceCheck(const ros::TimerEvent& e)
169  {
170  if(limited_distance_ * limited_distance_ <= getSqDistance())
171  {
172  ROS_INFO("Moved far enough, removing speed limit.");
173  //have to do this because a system call within a timer cb does not seem to play nice
174  if(remove_limit_thread_)
175  {
176  remove_limit_thread_->join();
177  delete remove_limit_thread_;
178  }
179  remove_limit_thread_ = new boost::thread(boost::bind(&MoveSlowAndClear::removeSpeedLimit, this));
180 
181  distance_check_timer_.stop();
182  }
183  }
