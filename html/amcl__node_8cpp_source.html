982 pf_vector_t
983 AmclNode::uniformPoseGenerator(void* arg)
984 {
985  map_t* map = (map_t*)arg;
986 #if NEW_UNIFORM_SAMPLING
987  unsigned int rand_index = drand48() * free_space_indices.size();
988  std::pair<int,int> free_point = free_space_indices[rand_index];
989  pf_vector_t p;
990  p.v[0] = MAP_WXGX(map, free_point.first);
991  p.v[1] = MAP_WYGY(map, free_point.second);
992  p.v[2] = drand48() * 2 * M_PI - M_PI;
993 #else
994  double min_x, max_x, min_y, max_y;
995 
996  min_x = (map->size_x * map->scale)/2.0 - map->origin_x;
997  max_x = (map->size_x * map->scale)/2.0 + map->origin_x;
998  min_y = (map->size_y * map->scale)/2.0 - map->origin_y;
999  max_y = (map->size_y * map->scale)/2.0 + map->origin_y;
1000 
1001  pf_vector_t p;
1002 
1003  ROS_DEBUG("Generating new uniform sample");
1004  for(;;)
1005  {
1006  p.v[0] = min_x + drand48() * (max_x - min_x);
1007  p.v[1] = min_y + drand48() * (max_y - min_y);
1008  p.v[2] = drand48() * 2 * M_PI - M_PI;
1009  // Check that it's a free cell
1010  int i,j;
1011  i = MAP_GXWX(map, p.v[0]);
1012  j = MAP_GYWY(map, p.v[1]);
1013  if(MAP_VALID(map,i,j) && (map->cells[MAP_INDEX(map,i,j)].occ_state == -1))
1014  break;
1015  }
1016 #endif
1017  return p;
1018 }

1065 
1066  // Do we have the base->base_laser Tx yet?
1067  if(frame_to_laser_.find(laser_scan->header.frame_id) == frame_to_laser_.end())
1068  {
1069  ROS_DEBUG("Setting up laser %d (frame_id=%s)\n", (int)frame_to_laser_.size(), laser_scan->header.frame_id.c_str());
1070  lasers_.push_back(new AMCLLaser(*laser_));
1071  lasers_update_.push_back(true);
1072  laser_index = frame_to_laser_.size();
1073

1075  tf::Vector3(0,0,0)),
1076  ros::Time(), laser_scan->header.frame_id);
1077  tf::Stamped<tf::Pose> laser_pose;
1078  try
1079  {
1080  this->tf_->transformPose(base_frame_id_, ident, laser_pose);
1081  }
1082  catch(tf::TransformException& e)
1083  {
1084  ROS_ERROR("Couldn't transform from %s to %s, "
1085  "even though the message notifier is in use",
1086  laser_scan->header.frame_id.c_str(),
1087  base_frame_id_.c_str());
1088  return;
1089  }
1090 
1091  pf_vector_t laser_pose_v;
1092  laser_pose_v.v[0] = laser_pose.getOrigin().x();
1093  laser_pose_v.v[1] = laser_pose.getOrigin().y();
1094  // laser mounting angle gets computed later -> set to 0 here!
1095  laser_pose_v.v[2] = 0;

1098  laser_pose_v.v[0],
1099  laser_pose_v.v[1],
1100  laser_pose_v.v[2]);
1101 
1102  frame_to_laser_[laser_scan->header.frame_id] = laser_index;
1103  } else {
1104  // we have the laser pose, retrieve laser index
1105  laser_index = frame_to_laser_[laser_scan->header.frame_id];
1106  }
1107 
1108  // Where was the robot when this scan was taken?
1109  pf_vector_t pose;
1110  if(!getOdomPose(latest_odom_pose_, pose.v[0], pose.v[1], pose.v[2],
1111  laser_scan->header.stamp, base_frame_id_))
1112  {
1113  ROS_ERROR("Couldn't determine robot's pose associated with laser scan");
1114  return;
1115  }
1116 
1117 
1118  pf_vector_t delta = pf_vector_zero();
1119 
1120  if(pf_init_)
1121  {
1122  // Compute change in pose
1123  //delta = pf_vector_coord_sub(pose, pf_odom_pose_);
1124  delta.v[0] = pose.v[0] - pf_odom_pose_.v[0];
1125  delta.v[1] = pose.v[1] - pf_odom_pose_.v[1];
1126  delta.v[2] = angle_diff(pose.v[2], pf_odom_pose_.v[2]);
1127 
1128  // See if we should update the filter
1129  bool update = fabs(delta.v[0]) > d_thresh_ ||
1130  fabs(delta.v[1]) > d_thresh_ ||
1131  fabs(delta.v[2]) > a_thresh_;
1132  update = update || m_force_update;
1133  m_force_update=false;
1134 
1135  // Set the laser update flags
1136  if(update)
1137  for(unsigned int i=0; i < lasers_update_.size(); i++)
1138  lasers_update_[i] = true;
1139  }
1140 
1141  bool force_publication = false;
1142  if(!pf_init_)
1143  {
1144  // Pose at last filter update
1145  pf_odom_pose_ = pose;
1146 
1147  // Filter is now initialized
1148  pf_init_ = true;
1149 
1150  // Should update sensor data
1151  for(unsigned int i=0; i < lasers_update_.size(); i++)
1152  lasers_update_[i] = true;
1153 
1154  force_publication = true;
1155 
1156  resample_count_ = 0;
1157  }
1158  // If the robot has moved, update the filter
1159  else if(pf_init_ && lasers_update_[laser_index])
1160  {
1161  //printf("pose\n");
1162  //pf_vector_fprintf(pose, stdout, "%.3f");
1163 
1164  AMCLOdomData odata;
1165  odata.pose = pose;
1166  // HACK
1167  // Modify the delta in the action data so the filter gets
1168  // updated correctly
1169  odata.delta = delta;
1170

1172  odom_->UpdateAction(pf_, (AMCLSensorData*)&odata);
1173 
1174  // Pose at last filter update
1175  //this->pf_odom_pose = pose;
1176  }
1177 
1178  bool resampled = false;
1179  // If the robot has moved, update the filter
1180  if(lasers_update_[laser_index])
1181  {
1182  AMCLLaserData ldata;
1183  ldata.sensor = lasers_[laser_index];
1184  ldata.range_count = laser_scan->ranges.size();
1185 
1186  // To account for lasers that are mounted upside-down, we determine the
1187  // min, max, and increment angles of the laser in the base frame.
1188  //
1189  // Construct min and max angles of laser, in the base_link frame.
1190  tf::Quaternion q;
1191  q.setRPY(0.0, 0.0, laser_scan->angle_min);
