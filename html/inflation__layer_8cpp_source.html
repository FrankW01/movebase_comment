126  double* min_y, double* max_x, double* max_y)
127 {

131  last_min_y_ = *min_y;
132  last_max_x_ = *max_x;
133  last_max_y_ = *max_y;
134  // For some reason when I make these -<double>::max() it does not
135  // work with Costmap2D::worldToMapEnforceBounds(), so I'm using
136  // -<float>::max() instead.
137  *min_x = -std::numeric_limits<float>::max();
138  *min_y = -std::numeric_limits<float>::max();
139  *max_x = std::numeric_limits<float>::max();
140  *max_y = std::numeric_limits<float>::max();
141  need_reinflation_ = false;
142  }
143  else
144  {
145  double tmp_min_x = last_min_x_;
146  double tmp_min_y = last_min_y_;
147  double tmp_max_x = last_max_x_;
148  double tmp_max_y = last_max_y_;
149  last_min_x_ = *min_x;
150  last_min_y_ = *min_y;
151  last_max_x_ = *max_x;
152  last_max_y_ = *max_y;
153  *min_x = std::min(tmp_min_x, *min_x) - inflation_radius_;
154  *min_y = std::min(tmp_min_y, *min_y) - inflation_radius_;
155  *max_x = std::max(tmp_max_x, *max_x) + inflation_radius_;
156  *max_y = std::max(tmp_max_y, *max_y) + inflation_radius_;
157  }

172 void InflationLayer::updateCosts(costmap_2d::Costmap2D& master_grid, int min_i, int min_j, int max_i,
173  int max_j)
174 {
175  boost::unique_lock < boost::recursive_mutex > lock(*inflation_access_);
176  if (!enabled_)
177  return;
178 
179  // make sure the inflation queue is empty at the beginning of the cycle (should always be true)
180  ROS_ASSERT_MSG(inflation_queue_.empty(), "The inflation queue must be empty at the beginning of inflation");
181 
182  unsigned char* master_array = master_grid.getCharMap();
183  unsigned int size_x = master_grid.getSizeInCellsX(), size_y = master_grid.getSizeInCellsY();
184 
185  if (seen_ == NULL) {
186  ROS_WARN("InflationLayer::updateCosts(): seen_ array is NULL");
187  seen_size_ = size_x * size_y;
188  seen_ = new bool[seen_size_];
189  }
190  else if (seen_size_ != size_x * size_y)
191  {
192  ROS_WARN("InflationLayer::updateCosts(): seen_ array size is wrong");
193  delete[] seen_;
194  seen_size_ = size_x * size_y;
195  seen_ = new bool[seen_size_];
196  }
197  memset(seen_, false, size_x * size_y * sizeof(bool));
198 
199  // We need to include in the inflation cells outside the bounding
200  // box min_i...max_j, by the amount cell_inflation_radius_. Cells
201  // up to that distance outside the box can still influence the costs
202  // stored in cells inside the box.
203  min_i -= cell_inflation_radius_;
204  min_j -= cell_inflation_radius_;
205  max_i += cell_inflation_radius_;
206  max_j += cell_inflation_radius_;
207 
208  min_i = std::max(0, min_i);
209  min_j = std::max(0, min_j);
210  max_i = std::min(int(size_x), max_i);
211  max_j = std::min(int(size_y), max_j);
212 
213  for (int j = min_j; j < max_j; j++)
214  {
215  for (int i = min_i; i < max_i; i++)
216  {
217  int index = master_grid.getIndex(i, j);
218  unsigned char cost = master_array[index];
219  if (cost == LETHAL_OBSTACLE)
220  {
221  enqueue(index, i, j, i, j);
222  }
223  }
224  }
225 
226  while (!inflation_queue_.empty())
227  {
228  // get the highest priority cell and pop it off the priority queue
229  const CellData& current_cell = inflation_queue_.top();
230 
231  unsigned int index = current_cell.index_;
232  unsigned int mx = current_cell.x_;
233  unsigned int my = current_cell.y_;
234  unsigned int sx = current_cell.src_x_;
235  unsigned int sy = current_cell.src_y_;
236 
237  // pop once we have our cell info
238  inflation_queue_.pop();
239 
240  // set the cost of the cell being inserted
241  if (seen_[index])
242  {
243  continue;
244  }
245 
246  seen_[index] = true;
247 
248  // assign the cost associated with the distance from an obstacle to the cell
249  unsigned char cost = costLookup(mx, my, sx, sy);
250  unsigned char old_cost = master_array[index];
251  if (old_cost == NO_INFORMATION && cost >= INSCRIBED_INFLATED_OBSTACLE)
252  master_array[index] = cost;
253  else
254  master_array[index] = std::max(old_cost, cost);
255 
256  // attempt to put the neighbors of the current cell onto the queue
257  if (mx > 0)
258  enqueue(index - 1, mx - 1, my, sx, sy);
259  if (my > 0)
260  enqueue(index - size_x, mx, my - 1, sx, sy);
261  if (mx < size_x - 1)
262  enqueue(index + 1, mx + 1, my, sx, sy);
263  if (my < size_y - 1)
264  enqueue(index + size_x, mx, my + 1, sx, sy);
265  }
266 }

277 inline void InflationLayer::enqueue(unsigned int index, unsigned int mx, unsigned int my,
278  unsigned int src_x, unsigned int src_y)
279 {
280  if (!seen_[index])
281  {
282  // we compute our distance table one cell further than the inflation radius dictates so we can make the check below
283  double distance = distanceLookup(mx, my, src_x, src_y);
284 
285  // we only want to put the cell in the queue if it is within the inflation radius of the obstacle point
286  if (distance > cell_inflation_radius_)
287  return;
288 
289  // push the cell data onto the queue and mark
290  CellData data(distance, index, mx, my, src_x, src_y);
291  inflation_queue_.push(data);
292  }

295 void InflationLayer::computeCaches()
296 {
297  if (cell_inflation_radius_ == 0)
298  return;
299 
300  // based on the inflation radius... compute distance and cost caches
301  if (cell_inflation_radius_ != cached_cell_inflation_radius_)
302  {
303  deleteKernels();
304 
305  cached_costs_ = new unsigned char*[cell_inflation_radius_ + 2];
306  cached_distances_ = new double*[cell_inflation_radius_ + 2];
307 
308  for (unsigned int i = 0; i <= cell_inflation_radius_ + 1; ++i)
309  {
310  cached_costs_[i] = new unsigned char[cell_inflation_radius_ + 2];
311  cached_distances_[i] = new double[cell_inflation_radius_ + 2];
312  for (unsigned int j = 0; j <= cell_inflation_radius_ + 1; ++j)
313  {
314  cached_distances_[i][j] = hypot(i, j);
315  }
316  }
317 
318  cached_cell_inflation_radius_ = cell_inflation_radius_;
319  }
320 
321  for (unsigned int i = 0; i <= cell_inflation_radius_ + 1; ++i)
322  {
323  for (unsigned int j = 0; j <= cell_inflation_radius_ + 1; ++j)
324  {
325  cached_costs_[i][j] = computeCost(cached_distances_[i][j]);
326  }
327  }
328 }

330 void InflationLayer::deleteKernels()
331 {
332  if (cached_distances_ != NULL)
333  {
334  for (unsigned int i = 0; i <= cached_cell_inflation_radius_ + 1; ++i)
335  {
336  if (cached_distances_[i])
337  delete[] cached_distances_[i];
338  }
339  if (cached_distances_)
340  delete[] cached_distances_;
341  cached_distances_ = NULL;
342  }
343 
344  if (cached_costs_ != NULL)
345  {
346  for (unsigned int i = 0; i <= cached_cell_inflation_radius_ + 1; ++i)
347  {
348  if (cached_costs_[i])
349  delete[] cached_costs_[i];
350  }
351  delete[] cached_costs_;
352  cached_costs_ = NULL;
353  }

356 void InflationLayer::setInflationParameters(double inflation_radius, double cost_scaling_factor)
357 {
358  if (weight_ != cost_scaling_factor || inflation_radius_ != inflation_radius)
359  {
360  // Lock here so that reconfiguring the inflation radius doesn't cause segfaults
361  // when accessing the cached arrays
362  boost::unique_lock < boost::recursive_mutex > lock(*inflation_access_);
363 
364  inflation_radius_ = inflation_radius;
365  cell_inflation_radius_ = cellDistance(inflation_radius_);
366  weight_ = cost_scaling_factor;
367  need_reinflation_ = true;
368  computeCaches();
369  }
370 }
