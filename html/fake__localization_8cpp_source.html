162  void stuffFilter(const nav_msgs::OdometryConstPtr& odom_msg){
163  //we have to do this to force the message filter to wait for transforms
164  //from odom_frame_id_ to base_frame_id_ to be available at time odom_msg.header.stamp
165  //really, the base_pose_ground_truth should come in with no frame_id b/c it doesn't make sense
166  boost::shared_ptr<nav_msgs::Odometry> stuff_msg(new nav_msgs::Odometry);
167  *stuff_msg = *odom_msg;
168  stuff_msg->header.frame_id = odom_frame_id_;
169  filter_->add(stuff_msg);
170  }

172  void update(const nav_msgs::OdometryConstPtr& message){
173  tf::Pose txi;
174  tf::poseMsgToTF(message->pose.pose, txi);
175  txi = m_offsetTf * txi;
176 
177  tf::Stamped<tf::Pose> odom_to_map;
178  try
179  {
180  m_tfListener->transformPose(odom_frame_id_, tf::Stamped<tf::Pose>(txi.inverse(), message->header.stamp, base_frame_id_), odom_to_map);
181  }
182  catch(tf::TransformException &e)
183  {
184  ROS_ERROR("Failed to transform to %s from %s: %s\n", odom_frame_id_.c_str(), base_frame_id_.c_str(), e.what());
185  return;
186  }
187  m_tfServer->sendTransform(tf::StampedTransform(odom_to_map.inverse(),
188  message->header.stamp + ros::Duration(transform_tolerance_),
189  global_frame_id_, message->header.frame_id));
190 
191  tf::Pose current;
192  tf::poseMsgToTF(message->pose.pose, current);
193 
194  //also apply the offset to the pose
195  current = m_offsetTf * current;
196 
197  geometry_msgs::Pose current_msg;
198  tf::poseTFToMsg(current, current_msg);
199 
200  // Publish localized pose
201  m_currentPos.header = message->header;
202  m_currentPos.header.frame_id = global_frame_id_;
203  m_currentPos.pose.pose = current_msg;
204  m_posePub.publish(m_currentPos);
205 
206  // The particle cloud is the current position. Quite convenient.
207  m_particleCloud.header = m_currentPos.header;
208  m_particleCloud.poses[0] = m_currentPos.pose.pose;
209  m_particlecloudPub.publish(m_particleCloud);
210  }

212  void initPoseReceived(const geometry_msgs::PoseWithCovarianceStampedConstPtr& msg){
213  tf::Pose pose;
214  tf::poseMsgToTF(msg->pose.pose, pose);
215 
216  if (msg->header.frame_id != global_frame_id_){
217  ROS_WARN("Frame ID of \"initialpose\" (%s) is different from the global frame %s", msg->header.frame_id.c_str(), global_frame_id_.c_str());
218  }
219 
220  // set offset so that current pose is set to "initialpose"
221  tf::StampedTransform baseInMap;
222  try{
223  // just get the latest
224  m_tfListener->lookupTransform(base_frame_id_, global_frame_id_, ros::Time(0), baseInMap);
225  } catch(tf::TransformException){
226  ROS_WARN("Failed to lookup transform!");
227  return;
228  }
229 
230  tf::Transform delta = pose * baseInMap;
231  m_offsetTf = delta * m_offsetTf;
232 
233  }
