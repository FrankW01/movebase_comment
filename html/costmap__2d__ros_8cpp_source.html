62 Costmap2DROS::Costmap2DROS(std::string name, tf::TransformListener& tf) :
63  layered_costmap_(NULL), name_(name), tf_(tf), stop_updates_(false), initialized_(true), stopped_(false),
64  robot_stopped_(false), map_update_thread_(NULL), last_publish_(0),
65  plugin_loader_("costmap_2d", "costmap_2d::Layer"), publisher_(NULL)
66 {
67  ros::NodeHandle private_nh("~/" + name);
68  ros::NodeHandle g_nh;
69 
70  // get our tf prefix
71  ros::NodeHandle prefix_nh;
72  std::string tf_prefix = tf::getPrefixParam(prefix_nh);
73 
74  // get two frames
75  private_nh.param("global_frame", global_frame_, std::string("/map"));
76  private_nh.param("robot_base_frame", robot_base_frame_, std::string("base_link"));
77 
78  // make sure that we set the frames appropriately based on the tf_prefix
79  global_frame_ = tf::resolve(tf_prefix, global_frame_);
80  robot_base_frame_ = tf::resolve(tf_prefix, robot_base_frame_);
81 
82  ros::Time last_error = ros::Time::now();
83  std::string tf_error;
84  // we need to make sure that the transform between the robot base frame and the global frame is available
85  while (ros::ok()
86  && !tf_.waitForTransform(global_frame_, robot_base_frame_, ros::Time(), ros::Duration(0.1), ros::Duration(0.01),
87  &tf_error))
88  {
89  ros::spinOnce();
90  if (last_error + ros::Duration(5.0) < ros::Time::now())
91  {
92  ROS_WARN("Timed out waiting for transform from %s to %s to become available before running costmap, tf error: %s",
93  robot_base_frame_.c_str(), global_frame_.c_str(), tf_error.c_str());
94  last_error = ros::Time::now();
95  }
96  // The error string will accumulate and errors will typically be the same, so the last
97  // will do for the warning above. Reset the string here to avoid accumulation.
98  tf_error.clear();
99  }
100 
101  // check if we want a rolling window version of the costmap
102  bool rolling_window, track_unknown_space, always_send_full_costmap;
103  private_nh.param("rolling_window", rolling_window, false);
104  private_nh.param("track_unknown_space", track_unknown_space, false);
105  private_nh.param("always_send_full_costmap", always_send_full_costmap, false);
106 
107  layered_costmap_ = new LayeredCostmap(global_frame_, rolling_window, track_unknown_space);
108 
109  if (!private_nh.hasParam("plugins"))
110  {
111  resetOldParameters(private_nh);
112  }
113 
114  if (private_nh.hasParam("plugins"))
115  {
116  XmlRpc::XmlRpcValue my_list;
117  private_nh.getParam("plugins", my_list);
118  for (int32_t i = 0; i < my_list.size(); ++i)
119  {
120  std::string pname = static_cast<std::string>(my_list[i]["name"]);
121  std::string type = static_cast<std::string>(my_list[i]["type"]);
122  ROS_INFO("Using plugin \"%s\"", pname.c_str());
123 
124  boost::shared_ptr<Layer> plugin = plugin_loader_.createInstance(type);
125  layered_costmap_->addPlugin(plugin);
126  plugin->initialize(layered_costmap_, name + "/" + pname, &tf_);
127  }
128  }
129 
130  // subscribe to the footprint topic
131  std::string topic_param, topic;
132  if (!private_nh.searchParam("footprint_topic", topic_param))
133  {
134  topic_param = "footprint_topic";
135  }
136 
137  private_nh.param(topic_param, topic, std::string("footprint"));
138  footprint_sub_ = private_nh.subscribe(topic, 1, &Costmap2DROS::setUnpaddedRobotFootprintPolygon, this);
139 
140  if (!private_nh.searchParam("published_footprint_topic", topic_param))
141  {
142  topic_param = "published_footprint";
143  }
144 
145  private_nh.param(topic_param, topic, std::string("oriented_footprint"));
146  footprint_pub_ = private_nh.advertise<geometry_msgs::PolygonStamped>("footprint", 1);
147 
148  setUnpaddedRobotFootprint(makeFootprintFromParams(private_nh));
149 
150  publisher_ = new Costmap2DPublisher(&private_nh, layered_costmap_->getCostmap(), global_frame_, "costmap",
151  always_send_full_costmap);
152 
153  // create a thread to handle updating the map
154  stop_updates_ = false;
155  initialized_ = true;
156  stopped_ = false;
157 
158  // Create a time r to check if the robot is moving
159  robot_stopped_ = false;
160  timer_ = private_nh.createTimer(ros::Duration(.1), &Costmap2DROS::movementCB, this);
161 
162  dsrv_ = new dynamic_reconfigure::Server<Costmap2DConfig>(ros::NodeHandle("~/" + name));
163  dynamic_reconfigure::Server<Costmap2DConfig>::CallbackType cb = boost::bind(&Costmap2DROS::reconfigureCB, this, _1,
164  _2);
165  dsrv_->setCallback(cb);
166 }

261 void Costmap2DROS::reconfigureCB(costmap_2d::Costmap2DConfig &config, uint32_t level)
262 {
263  transform_tolerance_ = config.transform_tolerance;
264  if (map_update_thread_ != NULL)
265  {
266  map_update_thread_shutdown_ = true;
267  map_update_thread_->join();
268  delete map_update_thread_;
269  }
270  map_update_thread_shutdown_ = false;
271  double map_update_frequency = config.update_frequency;
272 
273  double map_publish_frequency = config.publish_frequency;
274  if (map_publish_frequency > 0)
275  publish_cycle = ros::Duration(1 / map_publish_frequency);
276  else
277  publish_cycle = ros::Duration(-1);
278 
279  // find size parameters
280  double map_width_meters = config.width, map_height_meters = config.height, resolution = config.resolution, origin_x =
281  config.origin_x,
282  origin_y = config.origin_y;
283 
284  if (!layered_costmap_->isSizeLocked())
285  {
286  layered_costmap_->resizeMap((unsigned int)(map_width_meters / resolution),
287  (unsigned int)(map_height_meters / resolution), resolution, origin_x, origin_y);
288  }
289 
290  // If the padding has changed, call setUnpaddedRobotFootprint() to
291  // re-apply the padding.
292  if (footprint_padding_ != config.footprint_padding)
293  {
294  footprint_padding_ = config.footprint_padding;
295  setUnpaddedRobotFootprint(unpadded_footprint_);
296  }
297 
298  readFootprintFromConfig(config, old_config_);
299 
300  old_config_ = config;
301 
302  map_update_thread_ = new boost::thread(boost::bind(&Costmap2DROS::mapUpdateLoop, this, map_update_frequency));
303 }

305 void Costmap2DROS::readFootprintFromConfig(const costmap_2d::Costmap2DConfig &new_config,
306  const costmap_2d::Costmap2DConfig &old_config)
307 {
308  // Only change the footprint if footprint or robot_radius has
309  // changed. Otherwise we might overwrite a footprint sent on a
310  // topic by a dynamic_reconfigure call which was setting some other
311  // variable.
312  if (new_config.footprint == old_config.footprint &&
313  new_config.robot_radius == old_config.robot_radius)
314  {
315  return;
316  }
317 
318  if (new_config.footprint != "" && new_config.footprint != "[]")
319  {
320  std::vector<geometry_msgs::Point> new_footprint;
321  if (makeFootprintFromString(new_config.footprint, new_footprint))
322  {
323  setUnpaddedRobotFootprint(new_footprint);
324  }
325  else
326  {
327  ROS_ERROR("Invalid footprint string from dynamic reconfigure");
328  }
329  }
330  else
331  {
332  // robot_radius may be 0, but that must be intended at this point.
333  setUnpaddedRobotFootprint(makeFootprintFromRadius(new_config.robot_radius));
334  }
335 }

346 void Costmap2DROS::movementCB(const ros::TimerEvent &event)
347 {
348  // don't allow configuration to happen while this check occurs
349  // boost::recursive_mutex::scoped_lock mcl(configuration_mutex_);
350 
351  tf::Stamped < tf::Pose > new_pose;
352 
353  if (!getRobotPose(new_pose))
354  {
355  ROS_WARN_THROTTLE(1.0, "Could not get robot pose, cancelling reconfiguration");
356  robot_stopped_ = false;
357  }
358  // make sure that the robot is not moving
359  else if (fabs((old_pose_.getOrigin() - new_pose.getOrigin()).length()) < 1e-3
360  && fabs(old_pose_.getRotation().angle(new_pose.getRotation())) < 1e-3)
361  {
362  old_pose_ = new_pose;
363  robot_stopped_ = true;
364  }
365  else
366  {
367  old_pose_ = new_pose;
368  robot_stopped_ = false;
369  }
370 }

372 void Costmap2DROS::mapUpdateLoop(double frequency)
373 {
374  // the user might not want to run the loop every cycle
375  if (frequency == 0.0)
376  return;
377 
378  ros::NodeHandle nh;
379  ros::Rate r(frequency);
380  while (nh.ok() && !map_update_thread_shutdown_)
381  {
382  struct timeval start, end;
383  double start_t, end_t, t_diff;
384  gettimeofday(&start, NULL);
385 
386  updateMap();
387 
388  gettimeofday(&end, NULL);
389  start_t = start.tv_sec + double(start.tv_usec) / 1e6;
390  end_t = end.tv_sec + double(end.tv_usec) / 1e6;
391  t_diff = end_t - start_t;
392  ROS_DEBUG("Map update time: %.9f", t_diff);
393  if (publish_cycle.toSec() > 0 && layered_costmap_->isInitialized())
394  {
395  unsigned int x0, y0, xn, yn;
396  layered_costmap_->getBounds(&x0, &xn, &y0, &yn);
397  publisher_->updateBounds(x0, xn, y0, yn);
398 
399  ros::Time now = ros::Time::now();
400  if (last_publish_ + publish_cycle < now)
401  {
402  publisher_->publishCostmap();
403  last_publish_ = now;
404  }
405  }
406  r.sleep();
407  // make sure to sleep for the remainder of our cycle time
408  if (r.cycleTime() > ros::Duration(1 / frequency))
409  ROS_WARN("Map update loop missed its desired rate of %.4fHz... the loop actually took %.4f seconds", frequency,
410  r.cycleTime().toSec());
411  }
412 }

414 void Costmap2DROS::updateMap()
415 {
416  if (!stop_updates_)
417  {
418  // get global pose
419  tf::Stamped < tf::Pose > pose;
420  if (getRobotPose (pose))
421  {
422  double x = pose.getOrigin().x(),
423  y = pose.getOrigin().y(),
424  yaw = tf::getYaw(pose.getRotation());
425 
426  layered_costmap_->updateMap(x, y, yaw);
427 
428  geometry_msgs::PolygonStamped footprint;
429  footprint.header.frame_id = global_frame_;
430  footprint.header.stamp = ros::Time::now();
431  transformFootprint(x, y, yaw, padded_footprint_, footprint);
432  footprint_pub_.publish(footprint);
433 
434  initialized_ = true;
435  }
436  }

439 void Costmap2DROS::start()
440 {
441  std::vector < boost::shared_ptr<Layer> > *plugins = layered_costmap_->getPlugins();
442  // check if we're stopped or just paused
443  if (stopped_)
444  {
445  // if we're stopped we need to re-subscribe to topics
446  for (vector<boost::shared_ptr<Layer> >::iterator plugin = plugins->begin(); plugin != plugins->end();
447  ++plugin)
448  {
449  (*plugin)->activate();
450  }
451  stopped_ = false;
452  }
453  stop_updates_ = false;
454 
455  // block until the costmap is re-initialized.. meaning one update cycle has run
456  ros::Rate r(100.0);
457  while (ros::ok() && !initialized_)
458  r.sleep();
459 }

504 bool Costmap2DROS::getRobotPose(tf::Stamped<tf::Pose>& global_pose) const
505 {
506  global_pose.setIdentity();
507  tf::Stamped < tf::Pose > robot_pose;
508  robot_pose.setIdentity();
509  robot_pose.frame_id_ = robot_base_frame_;
510  robot_pose.stamp_ = ros::Time();
511  ros::Time current_time = ros::Time::now(); // save time for checking tf delay later
512 
513  // get the global pose of the robot
514  try
515  {
516  tf_.transformPose(global_frame_, robot_pose, global_pose);
517  }
518  catch (tf::LookupException& ex)
519  {
520  ROS_ERROR_THROTTLE(1.0, "No Transform available Error looking up robot pose: %s\n", ex.what());
521  return false;
522  }
523  catch (tf::ConnectivityException& ex)
524  {
525  ROS_ERROR_THROTTLE(1.0, "Connectivity Error looking up robot pose: %s\n", ex.what());
526  return false;
527  }
528  catch (tf::ExtrapolationException& ex)
529  {
530  ROS_ERROR_THROTTLE(1.0, "Extrapolation Error looking up robot pose: %s\n", ex.what());
531  return false;
532  }
533  // check global_pose timeout
534  if (current_time.toSec() - global_pose.stamp_.toSec() > transform_tolerance_)
535  {
536  ROS_WARN_THROTTLE(1.0,
537  "Costmap2DROS transform timeout. Current time: %.4f, global_pose stamp: %.4f, tolerance: %.4f",
538  current_time.toSec(), global_pose.stamp_.toSec(), transform_tolerance_);
539  return false;
540  }
541 
542  return true;
543 }
