114  inline unsigned char computeCost(double distance) const
115  {
116  unsigned char cost = 0;
117  if (distance == 0)
118  cost = LETHAL_OBSTACLE;
119  else if (distance * resolution_ <= inscribed_radius_)
120  cost = INSCRIBED_INFLATED_OBSTACLE;
121  else
122  {
123  // make sure cost falls off by Euclidean distance
124  double euclidean_distance = distance * resolution_;
125  double factor = exp(-1.0 * weight_ * (euclidean_distance - inscribed_radius_));
126  cost = (unsigned char)((INSCRIBED_INFLATED_OBSTACLE - 1) * factor);
127  }
128  return cost;
129  }
